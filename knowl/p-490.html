\begin{equation*} \begin{bmatrix}x\\y\\z\end{bmatrix} = \begin{bmatrix}2\\1\\3\end{bmatrix} + t\begin{bmatrix}-1\\0\\0\end{bmatrix}+s\begin{bmatrix}-1\\1\\-6\end{bmatrix}\text{.} \end{equation*}
